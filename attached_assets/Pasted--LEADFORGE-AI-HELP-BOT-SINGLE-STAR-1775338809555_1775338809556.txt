import React, { useEffect, useMemo, useRef, useState } from "react";

type ChatRole = "user" | "assistant";

type ChatMessage = {
  role: ChatRole;
  content: string;
};

type AppContext = {
  page:
    | "dashboard"
    | "project_creation"
    | "analytics"
    | "billing"
    | "settings"
    | "unknown";
  userName?: string;
  hasProjects: boolean;
  plan: "free" | "pro" | "business";
  recentActions: string[];
  industry?: string;
};

type ToolAction =
  | { type: "navigate"; page: string }
  | { type: "create_project"; payload?: { name?: string; industry?: string } }
  | { type: "open_upgrade" }
  | { type: "open_analytics" }
  | { type: "show_lead_tips" }
  | { type: "explain_current_page" }
  | { type: "suggest_next_step"; payload?: { step?: string } }
  | { type: "start_onboarding" }
  | { type: "build_ad"; payload?: { businessType?: string; city?: string } }
  | { type: "analyze_performance" };

type AIResponse = {
  reply: string;
  actions?: ToolAction[];
};

const quickButtons = [
  "Create my first project",
  "Help me get leads",
  "Explain this page",
  "Show analytics",
  "Upgrade plan",
];

function mapPathToPage(pathname: string): AppContext["page"] {
  const p = pathname.toLowerCase();
  if (p.includes("dashboard")) return "dashboard";
  if (p.includes("project") || p.includes("create")) return "project_creation";
  if (p.includes("analytics")) return "analytics";
  if (p.includes("billing") || p.includes("subscription")) return "billing";
  if (p.includes("settings")) return "settings";
  return "unknown";
}

export function AIHelpWidget(props: {
  userName?: string;
  hasProjects?: boolean;
  plan?: "free" | "pro" | "business";
  industry?: string;
  recentActions?: string[];
  onNavigate?: (page: string) => void;
  onCreateProject?: (payload?: { name?: string; industry?: string }) => Promise<void> | void;
  onOpenUpgrade?: () => void;
  onOpenAnalytics?: () => void;
  onShowLeadTips?: () => void;
  onExplainCurrentPage?: () => void;
  onStartOnboarding?: () => void;
  onBuildAd?: (payload?: { businessType?: string; city?: string }) => Promise<void> | void;
  onAnalyzePerformance?: () => Promise<void> | void;
}) {
  const [open, setOpen] = useState(false);
  const [loading, setLoading] = useState(false);
  const [input, setInput] = useState("");
  const [messages, setMessages] = useState<ChatMessage[]>([
    {
      role: "assistant",
      content:
        "Hi — I can help you use LeadForge, explain this page, suggest your next step, and take simple actions for you.",
    },
  ]);

  const bodyRef = useRef<HTMLDivElement | null>(null);

  const context: AppContext = useMemo(
    () => ({
      page: mapPathToPage(window.location.pathname),
      userName: props.userName || "there",
      hasProjects: props.hasProjects ?? false,
      plan: props.plan ?? "free",
      industry: props.industry || "general contractor",
      recentActions: props.recentActions ?? [],
    }),
    [props.userName, props.hasProjects, props.plan, props.industry, props.recentActions]
  );

  useEffect(() => {
    if (bodyRef.current) {
      bodyRef.current.scrollTop = bodyRef.current.scrollHeight;
    }
  }, [messages, open]);

  async function runAction(action: ToolAction) {
    switch (action.type) {
      case "navigate":
        props.onNavigate?.(action.page);
        break;
      case "create_project":
        await props.onCreateProject?.(action.payload);
        break;
      case "open_upgrade":
        props.onOpenUpgrade?.();
        break;
      case "open_analytics":
        props.onOpenAnalytics?.();
        break;
      case "show_lead_tips":
        props.onShowLeadTips?.();
        break;
      case "explain_current_page":
        props.onExplainCurrentPage?.();
        break;
      case "start_onboarding":
        props.onStartOnboarding?.();
        break;
      case "build_ad":
        await props.onBuildAd?.(action.payload);
        break;
      case "analyze_performance":
        await props.onAnalyzePerformance?.();
        break;
      case "suggest_next_step":
        // Optional UI hook if you want a toast/checklist step
        console.log("Next step suggestion:", action.payload?.step);
        break;
      default:
        break;
    }
  }

  async function sendMessage(text: string) {
    const trimmed = text.trim();
    if (!trimmed) return;

    const nextMessages = [...messages, { role: "user" as const, content: trimmed }];
    setMessages(nextMessages);
    setInput("");
    setLoading(true);

    try {
      const res = await fetch("/api/ai-help", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ messages: nextMessages, context }),
      });

      if (!res.ok) throw new Error("Failed to get AI response");

      const data: AIResponse = await res.json();

      setMessages((prev) => [...prev, { role: "assistant", content: data.reply }]);

      if (data.actions?.length) {
        for (const action of data.actions) {
          await runAction(action);
        }
      }
    } catch (err) {
      console.error(err);
      setMessages((prev) => [
        ...prev,
        {
          role: "assistant",
          content:
            "I hit a problem while trying to help. Please try again, or use one of the quick actions.",
        },
      ]);
    } finally {
      setLoading(false);
    }
  }

  return (
    <>
      <button
        onClick={() => setOpen((v) => !v)}
        style={{
          position: "fixed",
          bottom: 20,
          right: 20,
          borderRadius: 9999,
          padding: "14px 18px",
          border: "none",
          cursor: "pointer",
          fontWeight: 700,
          boxShadow: "0 10px 30px rgba(0,0,0,.18)",
          zIndex: 1000,
        }}
      >
        AI Help
      </button>

      {open && (
        <div
          style={{
            position: "fixed",
            bottom: 80,
            right: 20,
            width: 380,
            maxWidth: "calc(100vw - 24px)",
            height: 560,
            background: "#fff",
            borderRadius: 18,
            boxShadow: "0 20px 50px rgba(0,0,0,.18)",
            display: "flex",
            flexDirection: "column",
            overflow: "hidden",
            zIndex: 1000,
            border: "1px solid rgba(0,0,0,.08)",
          }}
        >
          <div style={{ padding: 16, borderBottom: "1px solid #eee" }}>
            <div style={{ fontSize: 18, fontWeight: 800 }}>AI Help</div>
            <div style={{ fontSize: 13, opacity: 0.7 }}>
              Ask anything or choose a quick action
            </div>
          </div>

          <div style={{ padding: 12, borderBottom: "1px solid #f2f2f2", display: "flex", gap: 8, flexWrap: "wrap" }}>
            {quickButtons.map((label) => (
              <button
                key={label}
                onClick={() => sendMessage(label)}
                style={{
                  padding: "8px 10px",
                  borderRadius: 9999,
                  border: "1px solid #ddd",
                  background: "#fafafa",
                  cursor: "pointer",
                  fontSize: 12,
                }}
              >
                {label}
              </button>
            ))}
          </div>

          <div
            ref={bodyRef}
            style={{
              flex: 1,
              overflowY: "auto",
              padding: 12,
              display: "flex",
              flexDirection: "column",
              gap: 10,
              background: "#fcfcfc",
            }}
          >
            {messages.map((m, i) => (
              <div
                key={`${m.role}-${i}`}
                style={{
                  alignSelf: m.role === "user" ? "flex-end" : "flex-start",
                  maxWidth: "85%",
                  padding: "10px 12px",
                  borderRadius: 14,
                  background: m.role === "user" ? "#111" : "#fff",
                  color: m.role === "user" ? "#fff" : "#111",
                  border: m.role === "assistant" ? "1px solid #ececec" : "none",
                  whiteSpace: "pre-wrap",
                  lineHeight: 1.4,
                  fontSize: 14,
                }}
              >
                {m.content}
              </div>
            ))}

            {loading && (
              <div
                style={{
                  alignSelf: "flex-start",
                  padding: "10px 12px",
                  borderRadius: 14,
                  background: "#fff",
                  border: "1px solid #ececec",
                  fontSize: 14,
                }}
              >
                Thinking...
              </div>
            )}
          </div>

          <form
            onSubmit={(e) => {
              e.preventDefault();
              sendMessage(input);
            }}
            style={{ padding: 12, borderTop: "1px solid #eee", display: "flex", gap: 8 }}
          >
            <input
              value={input}
              onChange={(e) => setInput(e.target.value)}
              placeholder="Ask how to use the app..."
              style={{
                flex: 1,
                padding: "12px 14px",
                borderRadius: 12,
                border: "1px solid #ddd",
                outline: "none",
              }}
            />
            <button
              type="submit"
              disabled={loading}
              style={{
                padding: "12px 14px",
                borderRadius: 12,
                border: "none",
                cursor: "pointer",
                fontWeight: 700,
              }}
            >
              Send
            </button>
          </form>
        </div>
      )}
    </>
  );
}
